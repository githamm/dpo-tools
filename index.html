            <!doctype html>
    <html>
     <head>        
           <!-- page title in line below -->
            <title>Bar Chart</title>
           <script src="http://extras.denverpost.com/frameworks/charts/chart.js"></script>
      </head>
      <body>
           <div style="width: 100%">
              <canvas id="canvas" height="350" width="600"></canvas>
           </div>

      <script>
     var barChartData = {
           // Enter labels for horizontal axis
            labels : ["January","February","March","April","May","June","July"],
         datasets : [
               {
                  fillColor : "rgba(72,99,160,0.5)", /* bar color */
                   strokeColor : "rgba(72,99,160,0.8)", /* bar border color */
                  highlightFill: "rgba(72,99,160,0.75)", /* bar color on hover/click */
                    highlightStroke: "rgba(72,99,160,1)", /* bar border color on hover/click */
                  data : ["1","3","5","7","9","11","13"] /* data */
                },
             // copy and paste above code for more bars; last set has no comma
              {
                  fillColor : "rgba(161,72,99,0.5)",
                   strokeColor : "rgba(161,72,99,0.8)",
                 highlightFill : "rgba(161,72,99,0.75)",
                  highlightStroke : "rgba(161,72,99,1)",
                   data : ["2","4","6","8","10","12","14"]
              },
             {
                  fillColor : "rgba(99,161,72,0.5)",
                   strokeColor : "rgba(99,161,72,0.8)",
                 highlightFill : "rgba(99,161,72,0.75)",
                  highlightStroke : "rgba(99,161,72,1)",
                   data : ["3","5","7","9","11","13","15"]
              }
          ]
      }
      window.onload = function(){
            var ctx = document.getElementById("canvas").getContext("2d");
          window.myBar = new Chart(ctx).Bar(barChartData, {
              responsive : true
          });
        }
      </script>
        </body>
    </html>
